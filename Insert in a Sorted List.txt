#include <bits/stdc++.h> 
using namespace std; 

struct Node
{
    int data;
    struct Node* next;
    
    Node(int x){
        data = x;
        next = NULL;
    }
};


void printList(Node* node) 
{ 
	while (node != NULL) { 
		cout << node->data <<" "; 
		node = node->next; 
	}  
	cout<<"\n";
}

/*
structure of the node of the list is as
struct Node
{
    int data;
    struct Node* next;

    Node(int x){
        data = x;
        next = NULL;
    }
};
*/

class Solution{
  public:
    // Should return head of the modified linked list
    Node *sortedInsert(struct Node* head, int data) {
        // Code here
       // Create the new node
        Node *newNode = new Node(data);

        // Check if the list is empty or if the new node should be inserted at the beginning
        if (head == NULL || data < head->data) {
            newNode->next = head;
            head = newNode;
            return head;
        }

        // Traverse the list to find the appropriate position to insert the new node
        Node *current = head;
        while (current->next != NULL && current->next->data < data) {
            current = current->next;
        }

        // Insert the new node
        newNode->next = current->next;
        current->next = newNode;

        return head;
        
    }
};
